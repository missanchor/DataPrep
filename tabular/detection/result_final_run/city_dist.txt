def distr_analysis_city_count(dirty_csv, attr_name):
    """

    - attr_name: The name of the attribute (should be 'city')
    
    Returns:
    - A string containing the detailed analysis results.
    """
    if attr_name != 'city':
        return "Attribute name does not match 'city'."
    
    city_counts = dirty_csv[attr_name].value_counts()

def distr_analysis_city_errors(dirty_csv, attr_name):
    """
    Analyzes cities with unexpected values (e.g., 'Unknown', 'errorval').

    Returns:

def distr_analysis_city_frequency(dirty_csv, attr_name):

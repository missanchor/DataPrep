def distr_analysis_city_consistency(dirty_csv, attr_name):
    # Ensure the attribute name is 'name'
    if attr_name != 'name':
        return "Attribute name is not 'name'."
    
    # Known cities
    known_cities = {'NY', 'LA', 'SF'}
    
    # Create a dictionary to map names to their respective cities
    name_to_city = {}
    for _, row in dirty_csv.iterrows():
        name = row[attr_name]
        city = row['city']
        if name not in name_to_city:
            name_to_city[name] = set()
        name_to_city[name].add(city)
    
    # Identify names with inconsistent city assignments
    inconsistent_names = []
    for name, cities in name_to_city.items():
        if len(cities) > 1 or any(city not in known_cities for city in cities):
            inconsistent_names.append(name)
    
    result = "Inconsistent city assignments found for the following names:\n"
    if inconsistent_names:
        result += "\n".join(inconsistent_names)
    else:
        result += "All names have consistent city assignments."
    
    return result

def distr_analysis_name_length(dirty_csv, attr_name):
    """
    Analyze the length distribution of names.
    """
    # Extract the length of each name
    name_lengths = dirty_csv[attr_name].str.len()
    
    # Basic statistics on name lengths
    min_length = name_lengths.min()
    max_length = name_lengths.max()
    mean_length = name_lengths.mean()
    median_length = name_lengths.median()
    
    # Representative names
    short_names = dirty_csv[name_lengths == min_length][attr_name].tolist()
    long_names = dirty_csv[name_lengths == max_length][attr_name].tolist()
    
    result = f"Name length distribution analysis:\n"
    result += f"Minimum length: {min_length} (Names: {short_names})\n"
    result += f"Maximum length: {max_length} (Names: {long_names})\n"
    result += f"Mean length: {mean_length:.2f}\n"
    result += f"Median length: {median_length:.2f}\n"
    return result

def distr_analysis_name_start(dirty_csv, attr_name):
    """
    Analyze the starting character distribution of names.
    """
    # Extract the first character of each name
    name_starts = dirty_csv[attr_name].str[0]
    
    # Basic statistics on starting characters
    start_counts = name_starts.value_counts()
    
    # Representative names
    common_starts = start_counts[start_counts > 1].index.tolist()
    uncommon_starts = start_counts[start_counts == 1].index.tolist()
    
    result = f"Starting character distribution analysis of names:\n"
    result += f"Common starting characters: {common_starts}\n"
    result += f"Uncommon starting characters: {uncommon_starts}\n"
    result += f"Counts:\n"
    for start, count in start_counts.items():
        result += f"  {start}: {count}\n"
    return result

def distr_analysis_name_frequency(dirty_csv, attr_name):

    Analyze the frequency distribution of names.

    max_count = name_counts.max()
    mean_count = name_counts.mean()
    median_count = name_counts.median()

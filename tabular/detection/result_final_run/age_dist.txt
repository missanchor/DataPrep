def distr_analysis_outliers(dirty_csv, attr_name):
    # Ensure the attribute name matches the column name
    if attr_name != 'age':
        return f"Attribute name does not match the expected column name 'age'. Found: {attr_name}"
    
    # Convert the 'age' column to numeric, coercing errors to NaN
    dirty_csv['age'] = pd.to_numeric(dirty_csv['age'], errors='coerce')
    
    # Drop rows with NaN values
    dirty_csv = dirty_csv.dropna(subset=['age'])
    
    if dirty_csv.empty:
        return "No valid age values found."
    
    Q1 = dirty_csv['age'].quantile(0.25)
    Q3 = dirty_csv['age'].quantile(0.75)
    IQR = Q3 - Q1
    
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    
    outliers = dirty_csv[(dirty_csv['age'] < lower_bound) | (dirty_csv['age'] > upper_bound)]
    
    result = (
        f"Outlier Detection for Age:\n"
        f"Lower Bound: {lower_bound:.2f}, Upper Bound: {upper_bound:.2f}\n"
        f"Outliers: {outliers['age'].tolist()}\n\n"
        f"Representative Values: {dirty_csv['age'].sample(n=5).tolist()}\n"
    )
    
    return result

def distr_analysis_age_groups(dirty_csv, attr_name):
    # Ensure the attribute name matches the column name
    if attr_name != 'age':
        return f"Attribute name does not match the expected column name 'age'. Found: {attr_name}"
    
    # Convert the 'age' column to numeric, coercing errors to NaN
    dirty_csv['age'] = pd.to_numeric(dirty_csv['age'], errors='coerce')
    
    # Drop rows with NaN values
    dirty_csv = dirty_csv.dropna(subset=['age'])
    
    if dirty_csv.empty:
        return "No valid age values found."
    
    # Define age groups
    bins = [0, 18, 25, 35, 45, 60, np.inf]
    labels = ['0-18', '19-25', '26-35', '36-45', '46-60', '60+']
    
    # Categorize ages
    dirty_csv['age_group'] = pd.cut(dirty_csv['age'], bins=bins, labels=labels, include_lowest=True)
    
    # Count occurrences in each group
    age_group_counts = dirty_csv['age_group'].value_counts().sort_index()
    
    result = (
        f"Age Group Analysis:\n"
        f"{age_group_counts}\n\n"
        f"Representative Values: {dirty_csv['age'].sample(n=5).tolist()}\n"
    )
    
    return result

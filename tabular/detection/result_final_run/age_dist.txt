def distr_analysis_age_distribution(dirty_csv, attr_name):
    # Ensure the column is numeric by converting strings to integers
    dirty_csv[attr_name] = dirty_csv[attr_name].str.replace(',', '').astype(int)
    
    # Describe the age distribution
    age_distribution = dirty_csv[attr_name].describe()
    
    # Detect outliers
    q1 = age_distribution['25%']
    q3 = age_distribution['75%']
    iqr = q3 - q1
    lower_bound = q1 - 1.5 * iqr
    upper_bound = q3 + 1.5 * iqr
    
    # Find and describe outliers
    outliers = dirty_csv[(dirty_csv[attr_name] < lower_bound) | (dirty_csv[attr_name] > upper_bound)]
    
    # Detailed analysis results
    analysis = (
        f"Age Distribution:\n"
        f"Count: {len(dirty_csv)}\n"
        f"Min: {age_distribution['min']}\n"
        f"Q1: {q1}\n"
        f"Median: {age_distribution['50%']}\n"
        f"Q3: {q3}\n"
        f"Max: {age_distribution['max']}\n"
        f"IQR: {iqr}\n"
        f"Lower Bound: {lower_bound}\n"
        f"Upper Bound: {upper_bound}\n"
        f"Outliers found: {len(outliers)}\n"
        f"Outliers: {outliers[attr_name].tolist()}\n"
    )
    
    return analysis

def distr_analysis_city_distribution(dirty_csv, attr_name):
    # Count the frequency of each city
    city_counts = dirty_csv['city'].value_counts()
    
    # Find the most common city
    most_common_city = city_counts.idxmax()
    most_common_count = city_counts.max()
    
    # Find the least common city
    least_common_city = city_counts.idxmin()
    least_common_count = city_counts.min()
    
    # Detailed analysis results
    analysis = (
        f"City Distribution:\n"
        f"Total Cities: {len(city_counts)}\n"
        f"Most Common City: {most_common_city} with {most_common_count} occurrences\n"
        f"Least Common City: {least_common_city} with {least_common_count} occurrences\n"
        f"City Counts:\n{city_counts}\n"
    )
    
    return analysis

def distr_analysis_name_length(dirty_csv, attr_name):
    # Calculate the length of each name
    name_lengths = dirty_csv['name'].str.len()
    
    # Describe the length distribution
    length_distribution = name_lengths.describe()
    
    # Find the longest and shortest name
    longest_name = dirty_csv['name'].str.len().idxmax()
    longest_length = dirty_csv['name'].str.len().max()
    
    shortest_name = dirty_csv['name'].str.len().idxmin()
    shortest_length = dirty_csv['name'].str.len().min()
    
    # Detailed analysis results
    analysis = (
        f"Name Length Distribution:\n"
        f"Count: {len(dirty_csv)}\n"
        f"Min Length: {shortest_length} (Name: {shortest_name})\n"
        f"Max Length: {longest_length} (Name: {longest_name})\n"
        f"Mean Length: {length_distribution['mean']}\n"
        f"Median Length: {length_distribution['50%']}\n"
        f"Std Dev Length: {length_distribution['std']}\n"
        f"Length Distribution:\n{length_distribution}\n"
    )
    
    return analysis

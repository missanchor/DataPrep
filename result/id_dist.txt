def distr_analysis_missing_values(dirty_csv, attr_name):
    if attr_name not in dirty_csv.columns:
        return f"Attribute {attr_name} does not exist in the dataset."
    
    missing_count = dirty_csv[attr_name].isnull().sum()
    missing_percentage = (missing_count / len(dirty_csv)) * 100
    
    result = f"Analysis of missing values in {attr_name}:\n"
    result += f"Total records: {len(dirty_csv)}\n"
    result += f"Missing records: {missing_count} ({missing_percentage:.2f}%)\n"
    result += f"Representative missing values: {dirty_csv[dirty_csv[attr_name].isnull()][attr_name].head(5).tolist()}"
    
    return result

def distr_analysis_outliers(dirty_csv, attr_name):
    if attr_name not in dirty_csv.columns or dirty_csv[attr_name].dtype != 'float64':
        return f"Attribute {attr_name} does not exist or is not numeric in the dataset."
    
    q1 = dirty_csv[attr_name].quantile(0.25)
    q3 = dirty_csv[attr_name].quantile(0.75)
    iqr = q3 - q1
    lower_bound = q1 - 1.5 * iqr
    upper_bound = q3 + 1.5 * iqr
    
    outliers = dirty_csv[(dirty_csv[attr_name] < lower_bound) | (dirty_csv[attr_name] > upper_bound)]
    
    result = f"Analysis of outliers in {attr_name}:\n"
    result += f"Lower bound: {lower_bound}\n"
    result += f"Upper bound: {upper_bound}\n"
    result += f"Number of outliers: {len(outliers)}\n"
    result += f"Representative outliers: {outliers[attr_name].head(5).tolist()}"
    
    return result
